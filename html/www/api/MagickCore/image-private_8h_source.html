56static inline size_t CastDoubleToLong(const double x)
57{
58 double
59 value;
60
61 if (IsNaN(x) != 0)
62 {
63 errno=ERANGE;
64 return(0);
65 }
66 if (x < 0.0)
67 {
68 value=ceil(x);
69 if (value < ((double) MAGICK_SSIZE_MIN))
70 {
71 errno=ERANGE;
72 return((ssize_t) MAGICK_SSIZE_MIN);
73 }
74 }
75 else
76 {
77 value=floor(x);
78 if (value > ((double) MAGICK_SSIZE_MAX))
79 {
80 errno=ERANGE;
81 return((ssize_t) MAGICK_SSIZE_MAX);
82 }
83 }
