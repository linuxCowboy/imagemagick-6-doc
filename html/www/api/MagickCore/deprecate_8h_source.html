372static inline double MagickEpsilonReciprocal(const double x)
373{
374 double
375 sign;
376
377 sign=x < 0.0 ? -1.0 : 1.0;
378 if ((sign*x) >= MagickEpsilon)
379 return(1.0/x);
380 return(sign/MagickEpsilon);
381}
